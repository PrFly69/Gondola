SAKH B 0 A_JumpIfInventory("IsFlying",1,"NoAmmo")
SAKH B 0 A_SpawnItemEx("SakugarneCrush",0,0,-10,0,0,-32)
SAKH B 0
SAKH B 0 ThrustThingZ(0,1,1,1)
SAKH B 1
SAKH B 0 A_CheckFloor("Jump")
loop
Jump1Spread:
SAKH AAA 0 A_SpawnItemEx("SakugarneDriller",0,0,16)
goto Jump+2
Jump:
SAKH A 0 A_JumpIfInventory("PowerSpread_ST",1,"Jump1Spread")
SAKH A 0 A_SpawnItemEx("SakugarneDriller",0,0,16)
SAKH A 0 A_JumpIfNoAmmo("NoAmmo")
SAKH A 0 A_JumpIfInventory("IsFlying",1,"NoAmmo")
SAKH A 0 A_PlaySoundEx("weapon/sakugarne","Weapon")
SAKH A 0 A_Stop
SAKH A 4
SAKH A 0 A_Stop
SAKH B 4
SAKH B 0 A_Stop
SAKH A 0 A_ChangeVelocity(17,0,16, CVF_RELATIVE)
SAKH C 5
Goto Air
Jump2Spread:
SAKH AAA 0 A_SpawnItemEx("SakugarneDriller",0,0,16)
goto Jump2+2
Jump2:
SAKH A 0 A_JumpIfInventory("PowerSpread_ST",1,"Jump2Spread")
SAKH A 0 A_SpawnItemEx("SakugarneDriller",0,0,16)
SAKH A 0 A_JumpIfNoAmmo("NoAmmo")
SAKH A 0 A_JumpIfInventory("IsFlying",1,"NoAmmo")
SAKH A 0 A_PlaySoundEx("weapon/sakugarne","Weapon")
SAKH A 0 A_Stop
SAKH A 3
SAKH A 0 A_Stop
SAKH B 3
SAKH B 0 A_Stop
SAKH A 0 A_ChangeVelocity(20,0,21, CVF_RELATIVE)
SAKH C 5
Goto Air

AltfireC:
SAKH A 0 A_TakeInventory("SakugarneActive",1)
goto AltFireS

Altfire:Goto AltfireCBM
AltfireCore:Goto Super::Altfire
ReadyC:Goto Ready+1
}
}
